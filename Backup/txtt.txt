    var map;
    var markers = [];
    var distanceElement = document.getElementById('distance');

    function initMap(userLat, userLng) {
    map = new google.maps.Map(document.getElementById('map'), {
        center: { lat: 14.4929, lng: 120.5049 },
        zoom: 12.5
    });

    // Fetched seller data, including lat and lng, from the PHP code
    var sellers = <?php echo json_encode($sellers); ?>;

    // Add the user's location as a marker on the map
    var userMarker = new google.maps.Marker({
        position: { lat: userLat, lng: userLng },
        map: map,
        title: 'User Location'
    });

    markers.push(userMarker);

    sellers.forEach(function (seller) {
        var marker = new google.maps.Marker({
            position: { lat: parseFloat(seller.lat), lng: parseFloat(seller.lng) },
            map: map,
            title: seller.shop_name,
        });

        markers.push(marker);
    });

    updateDistance(userLat, userLng);
}

function updateDistance(userLat, userLng) {
    for (var i = 1; i < markers.length; i++) {
        var sellerLatLng = markers[i].getPosition();
        var distance = google.maps.geometry.spherical.computeDistanceBetween(
            { lat: userLat, lng: userLng },
            sellerLatLng
        );

        // Convert meters to kilometers
        var distanceInKm = (distance / 1000).toFixed(2);

        // Update the distance in your table
        var distanceCell = document.createElement('td');
        distanceCell.textContent = distanceInKm + ' km';
        document.querySelectorAll('table.table tbody tr')[i - 1].appendChild(distanceCell);
    }
}

function updateUserLocation() {
    if (navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(function (position) {
            var userLocationElement = document.getElementById('user-location');
            userLocationElement.textContent = 'Your current location: ' + position.coords.latitude + ', ' + position.coords.longitude;

            // Call the function to calculate distances
            calculateDistances(position.coords.latitude, position.coords.longitude);
        }, function (error) {
            var userLocationElement = document.getElementById('user-location');
            userLocationElement.textContent = 'Failed to retrieve your location';
        });
    } else {
        var userLocationElement = document.getElementById('user-location');
        userLocationElement.textContent = 'Geolocation is not supported by your browser';
    }
}

window.onload = function () {
    updateUserLocation();
};
